Enregistrement Monome
    exposant    : entier
    coefficient : reel
    suivant     : Monome*
FinEnregistrement

Fonction creerMonome(coef : reel, exp : entier) : Monome*
Variables p : Monome*
Debut
    p <-- allouer mémoire pour un Monome
    p->exposant    <-- exp
    p->coefficient <-- coef
    p->suivant     <-- NULL
    retourner p
FinFonction

//Question 4 — insertion triée ordre décroissant
Procedure insererTrieMonome(*tete : Monome*, coef : reel, exp : entier)
Variables nouveau, p, precedent : Monome*
Debut
    nouveau <-- creerMonome(coef, exp)

    //Cas 1 : liste vide OU exposant plus grand que la tête
    Si *tete = NULL OU exp > (*tete)->exposant Alors
        nouveau->suivant <-- *tete
        *tete <-- nouveau

    //Cas 2 : exposant = 0 → toujours en fin de liste
    Sinon Si exp = 0 Alors
        precedent <-- *tete
        TantQue precedent->suivant != NULL Faire
            precedent <-- precedent->suivant
        FinTantQue
        precedent->suivant <-- nouveau

    //Cas 3 : chercher la bonne position
    Sinon
        precedent <-- *tete
        p         <-- (*tete)->suivant

        TantQue p != NULL ET p->exposant >= exp Faire
            precedent <-- p
            p         <-- p->suivant
        FinTantQue

        nouveau->suivant    <-- p
        precedent->suivant  <-- nouveau
    FinSi
FinProcedure

Procedure afficherPolynome(tete : Monome*)
Variables p      : Monome*
          premier : booleen
          coef    : reel
Debut
    Si tete = NULL Alors
        Afficher "0"
        retourner
    FinSi

    p       <-- tete
    premier <-- vrai

    TantQue p != NULL Faire
        coef <-- p->coefficient

        // Gestion du signe
        Si premier Alors
            Si coef < 0 Alors
                Afficher "-"
            FinSi
            premier <-- faux
        Sinon
            Si coef >= 0 Alors
                Afficher " + "
            Sinon
                Afficher " - "
            FinSi
        FinSi

        coef <-- valeur_absolue(coef)

        // Affichage selon l'exposant
        Si p->exposant = 0 Alors
            Afficher coef
        Sinon Si p->exposant = 1 Alors
            Si coef != 1 Alors
                Afficher coef, "*"
            FinSi
            Afficher "X"
        Sinon
            Si coef != 1 Alors
                Afficher coef, "*"
            FinSi
            Afficher "X^", p->exposant
        FinSi

        p <-- p->suivant
    FinTantQue
    Afficher retour à la ligne
FinProcedure
