Enregistrement Monome
    exposant    : entier
    coefficient : reel
    suivant     : Monome*
FinEnregistrement


Fonction creerMonome(coef : reel, exp : entier) : Monome*
Variables p : Monome*
Debut
    p <-- allouer mémoire pour un Monome
    p->exposant <-- exp
    p->coefficient <-- coef
    p->suivant <-- NULL
    retourner p
FinFonction


Procedure insererTrieMonome(*tete : Monome*, coef : reel, exp : entier)
Variables nouveau, p, precedent : Monome*
Debut
    nouveau <-- creerMonome(coef, exp)

    // Cas 1 : liste vide OU exposant plus grand que la tête
    Si *tete = NULL OU exp > (*tete)->exposant Alors
        nouveau->suivant <-- *tete
        *tete <-- nouveau

    // Cas 2 : exposant = 0 → toujours en fin de liste
    Sinon Si exp = 0 Alors
        precedent <-- *tete
        TantQue precedent->suivant != NULL Faire
            precedent <-- precedent->suivant
        FinTantQue
        precedent->suivant <-- nouveau

    // Cas 3 : chercher la bonne position
    Sinon
        precedent <-- *tete
        p <-- (*tete)->suivant
        TantQue p != NULL ET p->exposant >= exp Faire
            precedent <-- p
            p <-- p->suivant
        FinTantQue
        nouveau->suivant <-- p
        precedent->suivant <-- nouveau
    FinSi
FinProcedure


Procedure afficherPolynome(tete : Monome*)
Variables p : Monome*
          premier : booleen
          coef : reel
Debut
    Si tete = NULL Alors
        Afficher "0"
        retourner
    FinSi

    p <-- tete
    premier <-- vrai

    TantQue p != NULL Faire
        coef <-- p->coefficient

        Si premier Alors
            Si coef < 0 Alors
                Afficher "-"
            FinSi
            premier <-- faux
        Sinon
            Si coef >= 0 Alors
                Afficher " + "
            Sinon
                Afficher " - "
            FinSi
        FinSi

        coef <-- valeur_absolue(coef)

        Si p->exposant = 0 Alors
            Afficher coef
        Sinon Si p->exposant = 1 Alors
            Si coef != 1 Alors
                Afficher coef, "*"
            FinSi
            Afficher "X"
        Sinon
            Si coef != 1 Alors
                Afficher coef, "*"
            FinSi
            Afficher "X^", p->exposant
        FinSi

        p <-- p->suivant
    FinTantQue
    Afficher retour à la ligne
FinProcedure


Procedure skipEspaces(ch : chaine, *pos : entier)
Debut
    TantQue ch[*pos] = ' ' Faire
        *pos <-- *pos + 1
    FinTantQue
FinProcedure


Fonction analyseNaturel(ch : chaine, *pos : entier) : entier
Variables val : entier
Debut
    Si ch[*pos] n'est pas un chiffre Alors
        Afficher "ERREUR : chiffre attendu"
        Arrêter le programme
    FinSi
    val <-- 0
    TantQue ch[*pos] est un chiffre Faire
        val <-- val * 10 + (ch[*pos] - '0')
        *pos <-- *pos + 1
    FinTantQue
    retourner val
FinFonction


Fonction analyseNombre(ch : chaine, *pos : entier) : reel
Variables partieEntiere, partieDec, nbDecimales : reel
Debut
    partieEntiere <-- analyseNaturel(ch, pos)

    Si ch[*pos] = '.' Alors
        *pos <-- *pos + 1
        partieDec <-- 0
        nbDecimales <-- 0.1
        TantQue ch[*pos] est un chiffre Faire
            partieDec <-- partieDec + (ch[*pos] - '0') * nbDecimales
            nbDecimales <-- nbDecimales * 0.1
            *pos <-- *pos + 1
        FinTantQue
        retourner partieEntiere + partieDec
    FinSi

    retourner partieEntiere
FinFonction


Fonction analyseXpuissance(ch : chaine, *pos : entier) : entier
Debut
    Si ch[*pos] != 'X' Alors
        Afficher "ERREUR : 'X' attendu"
        Arrêter le programme
    FinSi
    *pos <-- *pos + 1

    Si ch[*pos] = '^' Alors
        *pos <-- *pos + 1
        retourner analyseNaturel(ch, pos)
    FinSi

    retourner 1
FinFonction


Procedure analyseMonome(ch : chaine, *pos : entier, *tete : Monome*, signe : entier)
Variables coef : reel
          expo : entier
Debut
    skipEspaces(ch, pos)
    coef <-- 0
    expo <-- 0

    Si ch[*pos] = 'X' Alors
        coef <-- 1
        expo <-- analyseXpuissance(ch, pos)

    Sinon Si ch[*pos] est un chiffre Alors
        coef <-- analyseNombre(ch, pos)
        skipEspaces(ch, pos)

        Si ch[*pos] = '*' Alors
            *pos <-- *pos + 1
            skipEspaces(ch, pos)
            expo <-- analyseXpuissance(ch, pos)
        Sinon
            expo <-- 0
        FinSi

    Sinon
        Afficher "ERREUR : monôme attendu"
        Arrêter le programme
    FinSi

    insererTrieMonome(tete, coef * signe, expo)
FinProcedure


Procedure analyserPolynome(ch : chaine, *tete : Monome*)
Variables pos : entier
          signe : entier
Debut
    pos <-- 0
    signe <-- 1

    skipEspaces(ch, &pos)

    // Signe optionnel devant le premier monôme
    Si ch[pos] = '-' Alors
        signe <-- -1
        pos <-- pos + 1
    FinSi

    // Lire le premier monôme (obligatoire)
    analyseMonome(ch, &pos, tete, signe)

    // Lire les monômes suivants
    TantQue ch[pos] != '\0' Faire
        skipEspaces(ch, &pos)

        Si ch[pos] = '+' Alors
            signe <-- 1
            pos <-- pos + 1
        Sinon Si ch[pos] = '-' Alors
            signe <-- -1
            pos <-- pos + 1
        Sinon
            Afficher "ERREUR : '+' ou '-' attendu"
            Arrêter le programme
        FinSi

        analyseMonome(ch, &pos, tete, signe)
    FinTantQue
FinProcedure

Variables chaine : chaine de caracteres

Debut
    tete <-- NULL
